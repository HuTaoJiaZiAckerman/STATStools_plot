# （1）转换数据，npy2txt
import numpy as np
import argparse

def npy_to_txt(input_file, output_file):
    try:
        # 直接使用np.load加载.npy文件
        data = np.load(input_file)
        
        # 确保输出文件名包含.txt扩展名
        if not output_file.endswith('.txt'):
            output_file += '.txt'
            
        # 正确使用np.savetxt，文件名作为第一个参数
        np.savetxt(output_file, data, fmt='%s')
        print(f"数据转换完成，已保存到 {output_file}")
    except Exception as e:
        print(f"转换过程中出错: {str(e)}")

# 设置命令行参数
if __name__ == '__main__':
    parser = argparse.ArgumentParser(description="将.npy文件转换为.txt文件")
    parser.add_argument('input_file', type=str, help="输入的.npy文件路径")
    parser.add_argument('output_file', type=str, help="输出的.txt文件路径或前缀")

    # 解析参数
    args = parser.parse_args()

    # 调用函数并传入参数
    npy_to_txt(args.input_file, args.output_file)

#! /usr/bin/env python3
# -*- coding: utf-8 -*-
# Author @caominghao
# Description @ split.chr.py
# CreateTime @ 2024-09-09 20:20:27
import numpy as np
import argparse

# 定义函数
def split_data(input_file, output_prefix):
    # 读取原始文件
    with open(input_file, 'r') as f:
        lines = f.readlines()

    # 将每一行转换为整数列表
    row1 = list(map(int, lines[0].split()))  # 第一行的数据
    row2 = list(map(int, lines[1].split()))  # 第二行的数据

    # 定义染色体区间
    chromosomes = {
        '1': (33, 3242598),
        '2': (3242599, 5450918),
        '3': (5450919, 7522105),
        '4': (7522106, 9458413),
        '5': (9458414, 11099887),
        '6': (11099888, 13554361),
        '7': (13554362, 15492155),
        '8': (15492156, 17585320),
        '9': (17585321, 19708162),
        '10': (19708163, 21128261),
        '11': (21128262, 22529530),
        '12': (22529531, 23646434),
        '13': (23646435, 26168207),
        '14': (26168208, 28213446),
        '15': (28213447, 30113327),
        '16': (30113328, 31446400),
        '17': (31446401, 32594118),
        '18': (32594119, 33600047)
    }

    # 将数据按照染色体区间拆分并保存
    for chr_name, (start, end) in chromosomes.items():
        # 切割第一行和第二行对应区间的数据
        chr_data_row1 = row1[start:end]
        chr_data_row2 = row2[start:end]

        # 保存为新文件
        with open(f'{output_prefix}_{chr_name}_output.txt', 'w') as f_out:
            for row in [chr_data_row1, chr_data_row2]:
                f_out.write(' '.join(map(str, row)) + '\n')

    print("数据拆分完成。")

# 设置命令行参数
if __name__ == '__main__':
    parser = argparse.ArgumentParser(description="根据染色体区间拆分数据文件")
    parser.add_argument('input_file', type=str, help="输入的 .txt 文件路径")
    parser.add_argument('output_prefix', type=str, help="输出文件的前缀名")

    # 解析参数
    args = parser.parse_args()

    # 调用函数并传入参数
    split_data(args.input_file, args.output_prefix)
